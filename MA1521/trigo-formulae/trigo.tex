    \section{reciprocal}
    \begin{align*}
            \csc A = \tfrac{1}{\sin A}
        \\* \sec A = \tfrac{1}{\cos A}
        \\* \cot A = \tfrac{1}{\tan A}
    \end{align*}

    \section{ periodicity}
    \begin{align*}
        \sin (A \pm 2\pi) = \sin A
        \\* \cos (A \pm 2\pi) = \cos A
        \\* \tan (A \pm \pi) = \tan A
    \end{align*}
\end{multicols}
    
\section{co-function}
\begin{align*}
    \sin (\pifrac{2} - A) = \cos A
    \\* \cos (\pifrac{2} - A) = \sin A
    \\* \tan (\pifrac{2} - A) = \cot A
\end{align*}



\section{sum \& difference}
\begin{align*}
        \sin (A \pm B) &= \sin A \cos B \pm \cos A \sin B
    \\* \cos (A \pm B) &= \cos A \cos B \mp \sin A \sin B
    \\* \tan (A \pm B) &= \frac{\tan A \pm \tan B}{1 \mp \tan A \tan B}
\end{align*}

\section{pythagorean identities}
\begin{align*}
        \sin^2A + \cos^2A &= 1
    \\* \sec^2A - \tan^2A &= 1
    \\* \csc^2A - \cot^2A &= 1
\end{align*}

\section{double angle}
\begin{align*}
        \sin (2A) &= 2\sin A \cos A
        \\* &= \frac{2\tan A}{1 + \tan^2A}
    \\* \cos (2A) &= \cos^2A - \sin^2A
    \\*         &= 1 - 2\sin^2A
    \\*         &= 2\cos^2A - 1
    \\*         &= \frac{1 - \tan^2A}{1 + \tan^2A}
    \\ \tan (2A) &= \frac{2\tan A}{1 - \tan^2A}
\end{align*}

\section{product to sum}
\begin{align*}
            \sin A \sin B &= \half \biggl[ \cos (A-B) - \cos (A+B) \biggr]
        \\* \cos A \cos B &= \half \biggl[ \cos (A-B) + \cos (A+B) \biggr]
        \\* \sin A \cos B &= \half \biggl[ \sin (A+B) + \sin (A-B) \biggr]
        \\* \cos A \sin B &= \half \biggl[ \sin (A+B) - \sin (A-B) \biggr]
\end{align*}

\section{sum to product}
\begin{align*}
        \sin A \pm \sin B &= 2 \sin \left(\frac{A \pm B}{2}\right) \cos \left(\frac{A \mp B}{2}\right)
    \\* \cos A + \cos B &= 2 \cos \left(\frac{A + B}{2}\right) \cos \left(\frac{A - B}{2}\right)
    \\* \cos A - \cos B &= -2 \sin \left(\frac{A + B}{2}\right) \sin \left(\frac{A - B}{2}\right)
\end{align*}
